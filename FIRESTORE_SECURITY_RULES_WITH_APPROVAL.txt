rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return request.auth.uid == userId;
    }
    
    function isCreatingOwn() {
      return request.resource.data.userId == request.auth.uid;
    }
    
    // User documents with approval system
    match /users/{userId} {
      // Users can read their own document
      allow read: if isAuthenticated() && isOwner(userId);
      
      // Only allow create during signup with approved: false
      allow create: if isAuthenticated() 
        && isOwner(userId)
        && request.resource.data.approved == false
        && request.resource.data.uid == request.auth.uid;
      
      // Users cannot update their own approval status
      // Users can update other fields (future profile updates)
      allow update: if isAuthenticated() 
        && isOwner(userId)
        && request.resource.data.approved == resource.data.approved;
    }
    
    // Notification documents for email extension
    match /notifications/{notificationId} {
      // Allow authenticated users to create notifications
      allow create: if isAuthenticated();
      // Email extension and admin need read/write access
      allow read, write: if true;
    }
    
    // Customers collection
    match /customers/{customerId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Credit Sales collection
    match /creditSales/{creditId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Payments collection
    match /payments/{paymentId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Settlements collection
    match /settlements/{settlementId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Sales collection
    match /sales/{saleId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Income/Expenses collection
    match /incomeExpenses/{recordId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Fuel Settings collection
    match /fuelSettings/{settingId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Settlement Types collection
    match /settlementTypes/{typeId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Income Categories collection
    match /incomeCategories/{categoryId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    
    // Expense Categories collection
    match /expenseCategories/{categoryId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && isCreatingOwn();
      allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
  }
}
